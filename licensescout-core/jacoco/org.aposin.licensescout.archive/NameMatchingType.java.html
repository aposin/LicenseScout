/**
 * Type of regular expression matching for archive names.
 * 
 * @see ArchiveIdentifier
 * @see PatternType
 *
 */
public enum NameMatchingType {

    /**
     * Name is matched exactly.
     */
    EXACT,
    /**
     * Name is matched as a regular expression.
     * 
     * @see PatternType
     */
    PATTERN;
}

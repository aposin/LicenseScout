16  package org.aposin.licensescout.core.test.util;
17  
18  import java.io.File;
19  import java.io.FileInputStream;
20  import java.io.FileOutputStream;
21  import java.io.IOException;
22  import java.nio.file.FileVisitResult;
23  import java.nio.file.Files;
24  import java.nio.file.Path;
25  import java.nio.file.SimpleFileVisitor;
26  import java.nio.file.attribute.BasicFileAttributes;
27  import java.util.ArrayList;
28  import java.util.List;
29  import java.util.zip.ZipEntry;
30  import java.util.zip.ZipOutputStream;
31  
32  import org.apache.commons.io.IOUtils;
33  
34  /**
35   * Utility methods for creating JAR files. 
36   */
37  public class CreateJarFileHelper {
38  
39      /**
40       * Collects a list of files by walking a file tree hierarchically.
41       * 
42       * <p>This method can be used to collect the files to package with the
43       * method {@link #createJarArchiveZip(File, List, Path)}.</p> 
44       * 
45       * @param startDirectory
46       * @return a list of files
47       * @throws IOException
48       */
49      public static List<File> collectFiles(final File startDirectory) throws IOException {
50          final List<File> files = new ArrayList<>();
51          final SimpleFileVisitor<Path> visitor = new SimpleFileVisitor<>() {
52  
53              public FileVisitResult visitFile(final Path file, final BasicFileAttributes attrs) throws IOException {
54                  files.add(file.toFile());
55                  return FileVisitResult.CONTINUE;
56              };
57          };
58          Files.walkFileTree(startDirectory.toPath(), visitor);
59          return files;
60      }
61  
62      /**
63       * Packages files into a ZIP/JAR file.
64       * 
65       * @param archiveFile the resulting ZIP file
66       * @param tobePackagedFiles list of files that should be packaged
67       * @param basePath the path that is 'subtracted' from the path of a file to package to obtain the relative path that is used
68       * as the entry name in the ZIP file
69       * @throws IOException
70       */
71      public static void createJarArchiveZip(final File archiveFile, final List<File> tobePackagedFiles,
72                                             final Path basePath)
73              throws IOException {
74          try (final FileOutputStream fileOutputstream = new FileOutputStream(archiveFile);
75                  final ZipOutputStream out = new ZipOutputStream(fileOutputstream)) {
76              for (final File tobePackaged : tobePackagedFiles) {
77                  if (tobePackaged == null || !tobePackaged.exists() || tobePackaged.isDirectory()) {
78                      continue;
79                  }
80                  final String relativePath = basePath.relativize(tobePackaged.toPath()).toString().replace('\\', '/');
81                  final ZipEntry entry = new ZipEntry(relativePath);
82                  entry.setTime(tobePackaged.lastModified());
83                  out.putNextEntry(entry);
84  
85                  // Write file to archive
86                  try (final FileInputStream in = new FileInputStream(tobePackaged)) {
87                      IOUtils.copy(in, out);
88                  }
89              }
90          }
91      }
92  }

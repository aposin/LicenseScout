16  package org.aposin.licensescout.license;
17  
18  import java.io.File;
19  import java.io.FileInputStream;
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.util.ArrayList;
23  import java.util.Collection;
24  import java.util.List;
25  
26  import org.aposin.licensescout.model.Notices;
27  import org.aposin.licensescout.model.Providers;
28  import org.aposin.licensescout.util.NullLog;
29  
30  /**
31   * Unit tests for {@link LicenseUtil}.
32   * 
33   * @see LicenseUtil
34   *
35   */
36  public class LicenseUtilTestUtil {
37  
38      private static final String CONFIGURATION_BASE_PATH = "src/test/resources/configuration/";
39      private static final String CHECKEDARCHIVES_PATH = CONFIGURATION_BASE_PATH + "checkedarchives.csv";
40      private static final String LICENSES_PATH = CONFIGURATION_BASE_PATH + "licenses.xml";
41  
42      /**
43       * Create {@link LicenseStoreData} for tests.
44       * @return an initialized {@link LicenseStoreData} object
45       * @throws Exception 
46       */
47      public static LicenseStoreData createLicenseStoreData() throws Exception {
48          final LicenseStoreData licenseStoreData = new LicenseStoreData();
49          final File licensesFile = new File(LICENSES_PATH);
50          try (final InputStream inputStream = new FileInputStream(licensesFile)) {
51              licenseStoreData.readLicenses(inputStream, new Notices(), false, new NullLog());
52          }
53          return licenseStoreData;
54      }
55  
56      /**
57       * @param licenseStoreData
58       * @return an object containing checkd archives information
59       * @throws IOException
60       */
61      public static LicenseCheckedList createLicenseCheckedList(final LicenseStoreData licenseStoreData)
62              throws IOException {
63          final LicenseCheckedList checkedArchives = new LicenseCheckedList();
64          final File checkedArchivesPathname = new File(CHECKEDARCHIVES_PATH);
65          try (final InputStream inputStream = new FileInputStream(checkedArchivesPathname)) {
66              checkedArchives.readCsv(inputStream, licenseStoreData, new Providers(), new Notices(), new NullLog());
67          }
68          return checkedArchives;
69      }
70  
71      /**
72       * @param licenseIdentifiers list of SPDX identifiers - may be empty
73       * @param licenseStoreData
74       * @return a list of licenses, maybe empty
75       */
76      public static List<License> createLicenseList(final String[] licenseIdentifiers,
77                                                    final LicenseStoreData licenseStoreData) {
78          final List<License> licenses = new ArrayList<>();
79          for (final String licenseIdentifier : licenseIdentifiers) {
80              final License license = licenseStoreData.getLicenseBySpdxIdentifier(licenseIdentifier);
81              if (license == null) {
82                  System.err.println("License not found in store: " + licenseIdentifier);
83              }
84              licenses.add(license);
85          }
86          return licenses;
87      }
88  
89      /**
90       * Compares two collections of licenses for equality.
91       * @param a one collection
92       * @param b another collection
93       * @return true if the collections are equal, false otherwise
94       */
95      public static boolean areEqual(final Collection<License> a, final Collection<License> b) {
96          boolean collectionsAreEqual = a.containsAll(b) && b.containsAll(a);
97          return collectionsAreEqual;
98      }

100 }

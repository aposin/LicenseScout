21  /**
22   * Test case for {@link CsvExporter}.
23   */
24  public class CsvExporterTest extends AbstractStringCompareExporterTest {
25  
26      /**
27       * {@inheritDoc}
28       */
29      @Override
30      protected IReportExporter createExporter() {
31          return CsvExporter.getInstance();
32      }
33  
34      /**
35       * {@inheritDoc}
36       */
37      @Override
38      protected OutputFileType getExpectedOutputFileType() {
39          return OutputFileType.CSV;
40      }
41  
42      /**
43       * Test method for {@link IReportExporter#export(org.aposin.licensescout.exporter.OutputResult, org.aposin.licensescout.exporter.ReportConfiguration)}.
44       * @throws Exception 
45       */
46      @Test
47      public void testExportEmptyArchiveListWithDocumentationUrl() throws Exception {
48          assertExport(TestVariant.EMPTY_WITH_DOCUMENTATION_URL_UTF_8, TemplateVariant.DEFAULT);
49      }
50  
51      /**
52       * Test method for {@link IReportExporter#export(org.aposin.licensescout.exporter.OutputResult, org.aposin.licensescout.exporter.ReportConfiguration)}.
53       * @throws Exception 
54       */
55      @Test
56      public void testExportWithArchiveListWithDocumentationUrl() throws Exception {
57          assertExport(TestVariant.ARCHIVE_WITH_DOCUMENTATION_URL_UTF_8, TemplateVariant.DEFAULT);
58      }
59  
60      /**
61       * {@inheritDoc}
62       */
63      @Override
64      protected String getOutputFilename() {
65          return "licensereport.csv";
66      }
67  
68      /**
69       * {@inheritDoc}
70       */
71      @Override
72      protected String getReferenceContent(final TestVariant testVariant) {
73          final StringBuffer refContent = new StringBuffer();
74          refContent.append(getReferenceHeader(testVariant.isWithDocumentationUrl()));
75          if (testVariant.isWithArchives()) {
76              refContent.append("\"JAVA\",\"fileName01\",\"version01\",\"\",\"DETECTED\",\"CONFLICTING\",\"path01\",");
77              if (testVariant.isWithDocumentationUrl()) {
78                  refContent.append("\"docUrl01\",");
79              }
80              refContent.append("\"spdxIdentifier11\"");
81              refContent.append(getNl());
82          }
83          return refContent.toString();
84      }
85  
86      private String getReferenceHeader(final boolean withDocumentationUrl) {
87          final String messageDigestAlgorithm = getMessageDigestAlgorithm();
88          final String nl = getNl();
89          final String referenceContent = "\"Type\",\"Filename\",\"Version\",\"Message Digest (" + messageDigestAlgorithm
90                  + ")\",\"Detection status\",\"Legal status\",\"Archive path\","
91                  + (withDocumentationUrl ? "\"Documentation\"," : "")
92                  + "\"License 1\",\"License 2\",\"License 3\",\"License 4\"" + nl;
93          return referenceContent;
94      }

96  }
